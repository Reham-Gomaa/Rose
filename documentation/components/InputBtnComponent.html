import { Component, EventEmitter, Input, Output } from "@angular/core";
// Angular Forms
import { FormsModule } from "@angular/forms";

@Component({
  selector: "app-input-btn",
  imports: [FormsModule],
  templateUrl: "./input-btn.component.html",
  styleUrl: "./input-btn.component.scss",
})
export class InputBtnComponent {
  @Input() placeholder = "";
  @Input() buttonText = "";
  @Input() type = "";
  @Input() name = "";
  @Input() showIcon = true;
  @Input() showBtn = true;
  @Input() inputValue = "";
  @Input() inputId = "input-field";
  @Input() buttonType = "submit";
  @Input() inputType = "text";
  @Input() disableMobileStyles = false;

  @Output() inputValueChange = new EventEmitter<string>();
  @Output() buttonClick = new EventEmitter<void>();

  onInputChange(newValue: string) {
    this.inputValue = newValue;
    this.inputValueChange.emit(newValue);
  }

  onButtonClick() {
    this.buttonClick.emit();
  }
}

<div class="input-button-wrapper" [class.disable-mobile]="disableMobileStyles">
  <input
    [id]="inputId"
    [type]="type"
    [name]="name"
    [placeholder]="placeholder"
    class="input-field"
    [ngModel]="inputValue"
    (ngModelChange)="onInputChange($event)"
    required
    aria-required="true"
  />
  @if (showBtn) {
  <button
    [type]="buttonType"
    class="input-button flex align-items-center justify-content-center"
    (click)="onButtonClick()"
  >
    {{ buttonText }}
    @if (showIcon) {
    <i class="pi pi-arrow-right px-2" aria-hidden="true"></i>
    }
  </button>
  }
</div>
